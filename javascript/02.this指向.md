# this指向

```javascript
var name = 22
var a = {
    name:111,
    say:function(){
        console.log(this.name)
    }
}
var fun = a.say;
fun(); // 22 fun.call(window,)
a.say(); // 111 a.asy.call(a,)
var b = {
    name:333,
    say:function(fun){
        fun() // fun.call(window)
    }
}
b.say(a.say); // 222 
b.say = a.say;
b.say(); // 333 (b.say === a.say === function(){console.log(this.name)}) === b.say.call(b,)
```
